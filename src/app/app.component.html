Generate UTC from parts

Moment

{{ generateUtcMoment }}
generateUtc(
  year: number, month: number, date: number,
  hour: number, minute: number, second: number
): _moment.Moment
    return this.moment.utc()
      .year(year).month(month).date(date)
      .hour(hour).minute(minute).second(second);
